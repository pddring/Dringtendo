// Caesar cipher (no wrapping around alphabet)
INP // input the caesar shift
STA Shift

CALL CaesarShift    // shift HELLO

LDA #33
STA StartPos
LDA #4
STA CharCount
CALL CaesarShift    // shift WORLD
HLT

CaesarShift: LDIR StartPos
LDA [CharCount]
ADD Shift
STA [CharCount]
LDA CharCount
SUB #1
STA CharCount
BRP CaesarShift
RET

StartPos: DAT 25
